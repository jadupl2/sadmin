----------------------------------------------------------------------------------------------------

----- IF PACKAGE IS INSTALL
  $ type who
  who is /usr/bin/who
  $ dpkg -S /usr/bin/who
  coreutils: /usr/bin/who


----- IF PACKAGE IS NOT INSTALL
root@raspi3w:~# type mail
-bash: type: mail: not found
root@raspi3w:~#  dpkg -S /usr/bin/mail
dpkg-query: no path found matching pattern /usr/bin/mail
root@raspi3w:~#

root@raspberrypi:~# type mail

root@raspberrypi:~# dpkg -S /usr/bin/mail.mailutils
